"""
Spawn 20 fake devices that register, then push encrypted logs
to the CoreNode every ~10 s (jittered). Requires encryption.key
generated by the Core server.
"""
import json, os, uuid, time, random, threading, requests
from cryptography.fernet import Fernet

CORE_URL = "http://localhost:8000"
KEY_FILE = "encryption.key"
NUM_SUBNODES = 20
PUSH_INTERVAL = 10  # seconds

cipher = Fernet(open(KEY_FILE, "rb").read())


def gen_log() -> dict:
    status = random.choices(["success", "error", "revoked"], [0.6, 0.3, 0.1])[0]
    return {
        "time": time.time(),
        "status": status,
        "output": "all good" if status == "success" else "meh",
    }


def register(device_name: str) -> str | None:
    r = requests.post(f"{CORE_URL}/api/register", json={"device_name": device_name})
    return r.json().get("node_id")


def push_logs(node_id: str, logs: list[dict]):
    payload = cipher.encrypt(json.dumps(logs).encode())
    try:
        r = requests.post(
            f"{CORE_URL}/api/upload_logs/{node_id}",
            data=payload,
            headers={"Content-Type": "application/octet-stream"},
            timeout=5,
        )
        print(f"[{node_id[:6]}] push {r.status_code}")
    except Exception as e:
        print(f"[{node_id[:6]}] push error: {e}")


def simulate(device_name: str):
    nid = register(device_name)
    if not nid:
        print(f"{device_name} failed to register")
        return
    while True:
        push_logs(nid, [gen_log() for _ in range(random.randint(1, 5))])
        time.sleep(PUSH_INTERVAL + random.uniform(-2, 2))


if __name__ == "__main__":
    for _ in range(NUM_SUBNODES):
        name = f"Mock-{uuid.uuid4().hex[:8]}"
        threading.Thread(target=simulate, args=(name,), daemon=True).start()
    while True:
        time.sleep(60)
