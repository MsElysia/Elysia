"""
Minimal real client:
• Loads node_id + core_url from config.json (scaffolds if missing)
• Sends a single encrypted “boot” heartbeat log on start
"""
import os, json, time, requests
from cryptography.fernet import Fernet

CFG_FILE = "config.json"
KEY_FILE = "encryption.key"

if not os.path.exists(CFG_FILE):
    json.dump(
        {"node_id": "REPLACE_ME", "core_url": "http://localhost:8000"},
        open(CFG_FILE, "w"),
        indent=2,
    )
    raise SystemExit("✍️  Edit config.json with your real node_id first.")

cfg = json.load(open(CFG_FILE))
cipher = Fernet(open(KEY_FILE, "rb").read())

log = [
    {
        "time": time.time(),
        "status": "boot",
        "output": "mobile heartbeat",
    }
]

payload = cipher.encrypt(json.dumps(log).encode())
url = f"{cfg['core_url']}/api/upload_logs/{cfg['node_id']}"
r = requests.post(url, data=payload, headers={"Content-Type": "application/octet-stream"})
print("Push →", r.status_code, r.text)
